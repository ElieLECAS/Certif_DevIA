alembic upgrade head


# Commandes pour builder l'image Docker et lancer la production

# 1. Build de l'image Docker (à lancer dans le dossier racine du projet)
docker build -t monrepo/fastapi-prod:latest -f Dockerfile .

# 2. Lancer la production (dans le dossier racine ou E5)
docker-compose -f docker-compose.prod.yml up -d
